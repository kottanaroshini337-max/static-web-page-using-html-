<!DOCTYPE html>
<html lang="en">


<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Simple CSS Example</title>


    <style>
        /* Target the layout box */
        .card {
            /* 1. Dimensions and Sizing */
            width: 300px;


            /* 2. Inner Space & Outer Space (The Box Model) */
            padding: 20px;
            margin: 40px auto;
            /* Centers the box on the screen */


            /* 3. Surface & Background Visuals */
            background-color: #ffffff;
            border: 2px solid #3498db;
            /* Solid blue frame line */
            border-radius: 8px;
            /* Softens and rounds the corners */


            /* 4. Typography Rules */
            font-family: Arial, sans-serif;
            text-align: center;
            /* Centers all text inside the box */
            box-shadow: 0 4px 10px rgba(0, 0, 0, 0.1);
        }


        /* Target the button inside the box */
        .btn {
            /* 5. Color Formats */
            background-color: #2ecc71;
            /* Vibrant green fill */
            color: white;
            /* White text */


            border: none;
            padding: 10px 20px;
            border-radius: 4px;
            font-weight: bold;
            cursor: pointer;
            /* Changes mouse pointer to a hand icon */
        }
    </style>
</head>
